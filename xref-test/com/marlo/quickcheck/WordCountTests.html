16  /** WordCountUtil tests using both Jupiter (JUnit5) and junit-quickcheck. */
17  @RunWith(JUnitQuickcheck.class)
18  public class WordCountTests {
19  
20    /** Word count using traditional JUnit. */
21    @Test
22    public void testWordCounts() {
23      assertEquals(3, WordCountUtils.count(new Scanner("one two three")));
24      assertEquals(3, WordCountUtils.count(Stream.of("one two three")));
25    }
26  
27    /** Test empty word count (i.e. 0 expected) using JUnit. */
28    @Test
29    public void testEmpty() {
30      assertEquals(0, WordCountUtils.count(new Scanner("")));
31      assertEquals(0, WordCountUtils.count(Stream.of("")));
32    }
33  
34    /**
35     * Test scanner with empty word count (0 expected). Trails increased from the default of 100 to
36     * 1000.
37     */
38    @Property(trials = 1000)
39    public void testWhitespaceScannerGenerator(
40        final @From(WhiteSpaceGenerator.class) String whitespaces) {
41      assertEquals(0, whitespaces.trim().length());
42      assertEquals(0, WordCountUtils.count(new Scanner(whitespaces)));
43    }
44  
45    /**
46     * Test stream with empty word count (0 expected). Trails increased from the default of 100 to
47     * 1000.
48     */
49    @Property(trials = 1000)
50    public void testWhitespaceStreamGenerator(
51        final @From(WhiteSpaceGenerator.class) String whitespaces) {
52      assertEquals(0, whitespaces.trim().length());
53      assertEquals(0, WordCountUtils.count(Stream.of(whitespaces)));
54    }
55  
56    /**
57     * Count words from a random string and compare with naive word count algorithm.
58     *
59     * @param words a word from random string
60     */
61    @Property
62    public void testCountUsingSplit(final String words) {
63      assertNotNull(words);
64      assertFalse(words.trim().isEmpty());
65      final long count = WordCountUtils.count(Stream.of(words));
66      assertEquals(words.trim().split("\\s+").length, count);
67    }

82    /**
83     * Test word counter is same for stream as scanner. <br>
84     *
85     * @param words a list of words from random string
86     */
87    @Property
88    public void testCountStreamSameAsScanner(final List<String> words) {
89      final String sentence = String.join("", words);
90      assertEquals(
91          WordCountUtils.count(Stream.of(sentence)), WordCountUtils.count(new Scanner(sentence)));
92    }
93  
94    /**
95     * Test a "sentence" of alphanumeric words. A sentence is a list of words separated by a space.
96     *
97     * @param words build a sentence from a word stream
98     */
99    @Property
100   public void testAlphanumericSentence(
101       final List<@From(AlphaNumericGenerator.class) String> words) {
102     final String sentence = String.join(" ", words);
103     assertEquals(
104         WordCountUtils.count(new Scanner(sentence)), WordCountUtils.count(Stream.of(sentence)));
105   }

107   /**
108    * Test alphanumeric word is same for stream as scanner using Alphanumeric generator. Trials
109    * increased from the default of 100 to 1000.
110    *
111    * @param word a random alphanumeric word
112    */
113   @Property(trials = 1000)
114   public void testAlphanumericWord(final @From(AlphaNumericGenerator.class) String word) {
115     assertEquals(1, WordCountUtils.count(new Scanner(word)));
116     assertEquals(1, WordCountUtils.count(Stream.of(word)));
