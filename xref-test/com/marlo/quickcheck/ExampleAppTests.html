19    /** Main using custom STDIN. */
20    @Test
21    public void testMainGoodInput() throws IOException {
22      // override stdin
23      final InputStream originalInStream = System.in;
24      final InputStream testInStream =
25          new ByteArrayInputStream("one two".getBytes(StandardCharsets.UTF_8));
26      // capture stdout to byte array
27      final PrintStream originalOutStream = System.out;
28      final ByteArrayOutputStream testByteStream = new ByteArrayOutputStream();
29      final PrintStream testOutStream = new PrintStream(testByteStream, true, StandardCharsets.UTF_8);
30      try {
31        // redirect stdin and stdout
32        System.setIn(testInStream);
33        System.setOut(testOutStream);
34        // call main
35        ExampleApp.main(new String[] {});
36        // inspect stdout
37        assertEquals("2\n", testByteStream.toString(StandardCharsets.UTF_8));
38      } finally {
39        // restore original stdin and stdout
40        testInStream.close();
41        testOutStream.close();
42        testByteStream.close();
43        System.setIn(originalInStream);
44        System.setOut(originalOutStream);
45      }
46    }
47  
48    /** Main does not accept parameters. */
49    @Test
50    public void testMainBadInput() throws IOException {
51      // override stdin
52      final InputStream originalInStream = System.in;
53      final InputStream testInStream = new ByteArrayInputStream(new byte[0]);
54      // capture stdout to byte array
55      final PrintStream originalErrStream = System.err;
56      final ByteArrayOutputStream testByteStream = new ByteArrayOutputStream();
57      final PrintStream testErrStream = new PrintStream(testByteStream, true, StandardCharsets.UTF_8);
58      try {
59        // redirect stdio
60        System.setIn(testInStream);
61        System.setErr(testErrStream);
62        // call my main
63        ExampleApp.main(new String[] {"bad args"});
64        // inspect stdout
65        assertEquals(
66            "USAGE: count words from file read from STDIN",
67            testByteStream.toString(StandardCharsets.UTF_8));
68      } finally {
69        // restore original stdin and stdout
70        testInStream.close();
71        testErrStream.close();
72        System.setIn(originalInStream);
73        System.setErr(originalErrStream);
74      }
75    }
